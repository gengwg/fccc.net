function db_run($query) {
	$db = mysql_connect('10.8.11.133', 'hymns', 'nm!v84m#Jg');
	mysql_select_db('hymns');
	$res = mysql_query($query);
	if (!$res) {
		die('Invalid query: ' . mysql_error());
	} else {
		$rows = array();
		while ($row = mysql_fetch_assoc($res)) {
			array_push($rows, $row);
		}
	}
	mysql_close($db);
	return $rows;
}
